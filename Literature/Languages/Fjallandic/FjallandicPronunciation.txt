[[PRONUNCATION]]

[[MONOPHTHONGS]]
	A: /ɑ/
	E: /ɛ/
	I: /ɪ/
	Í: /iː/
	O: /o/ or /ɔ/, more commonly /o/
	Ö: /œː/
	U: /ʊ/
	Ú: /uː/
	Y: /y/

[[DIPHTHONGS]]
	Á: /ɑuː/
	Æ: /aiː/
	AU: /œyː/
	ÁI: /ɑiː/
	É: /jɛː/
	EI, EY: /ɛiː/
	Ó: /ouː/
	Ý: /yɪː/, or /yː/ if the next IPA is /j/
	JA: /jaː/

[[CONSONANTS]]
	B: /b/
	D: /d/
	Ð: /ð/
	F: /f/
	G: /g/
	G that follows an Æ, É, I, Í, Ö, or the combination AU: /ɣ/
		Example: ægi is pronounced /'ai.ɣɪ/, and lög is pronounced /lœɣ/, but getr is pronounced /gɛtr/
	H: /h/
	J: /j/
	K: /k/
	L: /l/ or /l̪/. Examples: bolli is /bolɪ/, bolr is /bolr/, allt is /ɑl̪t/, eldr is /ɛl̪dr/, mel is /mɛl̪/
	-FL: /fl̪/
	-GL: /kx/ Example: tagl is /tɑkx/
	-RL: /rɫ/
	-SL: /sɫ/
	M: /m/
	N: /n/
	P: /p/
	R: /r/
	S: /s/
	T: /t/
	V: /ʋ/
	HV: /ʍ/
	KV: /kf/
	X: /ks/
	Z: /z/
	Þ: /θ/

	-NG: /ŋk/
	-NGR: /ŋgr/

	Stress: Defaulted to the first syllable of the word.
